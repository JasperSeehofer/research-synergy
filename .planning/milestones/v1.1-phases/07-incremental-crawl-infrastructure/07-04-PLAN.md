Purpose: The arxiv-rs library returns `pdf_url = ""` for some papers. Without a guard, this produces a reqwest "builder error" that silently kills reference fetching for every paper the new crawl loop processes. This unblocks UAT Test 1 and the five tests that depend on it.

Output: Updated `arxiv_utils.rs` with guard + fallback. Unit tests for the new behavior.
</objective>

fn test_html_url_fallback_for_empty_pdf_url() {
    // Validates the fallback logic directly
    let paper_id = "2503.18887";
    let pdf_url = "";
    let html_url = if pdf_url.is_empty() {
        format!("https://arxiv.org/html/{}", paper_id)
    } else {

    assert_eq!(html_url, "https://arxiv.org/html/2503.18887");

    assert_eq!(html_url, "https://arxiv.org/html/2503.18887");
