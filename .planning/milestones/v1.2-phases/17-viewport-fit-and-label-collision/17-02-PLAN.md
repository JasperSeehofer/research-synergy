        self.ctx.set_font("11px monospace");

        if let Some(ref cache) = self.label_cache {
            use crate::graph::label_collision::{PILL_HEIGHT, PILL_H_PAD, LABEL_NODE_GAP, PILL_CORNER_RADIUS};

            for &i in &cache.visible_indices {

                let text_w = cache.text_widths[i];

                self.ctx.round_rect_with_f64(label_x, label_y, pill_w, PILL_HEIGHT, PILL_CORNER_RADIUS).unwrap();

                self.ctx.set_line_width(1.0);
                self.ctx.stroke();

                self.ctx.set_fill_style_str("#cccccc");
                self.ctx.fill_text(&node.label(), label_x + PILL_H_PAD, label_y + 14.0).unwrap();
