   ```rust
   // Cluster recompute (D-09): every 10 frames during simulation, or on first settle
   if s.graph.label_mode == LabelMode::Keywords {
       s.cluster_frame_counter += 1;
       let should_recompute = s.cluster_result.is_none()
           || (s.graph.simulation_running && s.cluster_frame_counter % 10 == 0)
           || just_converged;  // just_converged is the bool from check_alpha_convergence()
       if should_recompute && s.graph.alpha < 0.15 {
           // D-07: defer until simulation has partially settled (alpha < 0.15)
           let positions: Vec<(f64, f64)> = s.graph.nodes.iter()
               .filter(|n| n.lod_visible && n.temporal_visible)
               .map(|n| (n.x, n.y))
               .collect();
           let keywords: Vec<Vec<(String, f32)>> = s.graph.nodes.iter()
               .filter(|n| n.lod_visible && n.temporal_visible)
               .map(|n| n.top_keywords.clone())
               .collect();
           if positions.len() >= 3 {
               s.cluster_result = Some(crate::graph::kmeans::compute_clusters(&positions, &keywords));
           }
       }
   } else {
       s.cluster_result = None;
   }
   ```
