```rust
use axum::{Router, routing::{get, post}};
use clap::Args;
use leptos_axum::handle_server_fns_with_context;
use leptos::prelude::*;
use resyn_core::database::client::connect_local;
use std::sync::Arc;
use tracing::info;

#[derive(Args, Debug)]
pub struct ServeArgs {
    /// Database connection string
    #[arg(long, default_value = "surrealkv://./data")]
    pub db: String,

    /// Server port
    #[arg(long, default_value_t = 3000)]
    pub port: u16,
}

pub async fn run(args: ServeArgs) -> anyhow::Result<()> {
    let db = connect_local(&args.db).await?;
    let db = Arc::new(db);

    let app = Router::new()
        .route("/api/{tail:.*}", post({
            let db = db.clone();
            move |req: axum::http::Request<axum::body::Body>| {
                let db = db.clone();
                async move {
                    handle_server_fns_with_context(
                        move || {
                            provide_context(db.clone());
                        },
                        req,
                    ).await
                }
            }
        }))
        .fallback_service(
            tower_http::services::ServeDir::new("../resyn-app/dist")
                .fallback(tower_http::services::ServeFile::new("../resyn-app/dist/index.html"))
        );

    let addr = format!("127.0.0.1:{}", args.port);
    info!("ReSyn server listening on http://{}", addr);
    let listener = tokio::net::TcpListener::bind(&addr).await?;
    axum::serve(listener, app).await?;
    Ok(())
}
```

The executor should consult `docs.rs/leptos_axum/0.8` if the above pattern doesn't compile and adapt accordingly — the intent is: wildcard POST route at `/api/*`, context provides `Arc<Db>`.
