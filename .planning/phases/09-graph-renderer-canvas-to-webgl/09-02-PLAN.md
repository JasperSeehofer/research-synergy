    2. **Create `resyn-worker/src/forces.rs`:**

       Constants:
       ```rust
       pub const REPULSION_STRENGTH: f64 = -30.0;
       pub const ATTRACTION_STRENGTH: f64 = 0.1;
       pub const CENTER_GRAVITY: f64 = 0.05;
       pub const ALPHA_DECAY: f64 = 0.92;
       pub const ALPHA_MIN: f64 = 0.001;
       pub const THETA: f64 = 0.9;
       ```

       `pub fn simulation_tick(nodes: &mut [NodeData], edges: &[(usize, usize)], alpha: &mut f64) -> bool`:
       - Build quadtree from current positions
       - For each non-pinned node: compute Barnes-Hut repulsion, edge attraction (Hooke's law: force = ATTRACTION_STRENGTH * (distance - ideal_distance)), center gravity (pull toward 0,0 with strength CENTER_GRAVITY)
       - Apply velocity: vx += fx * alpha, vy += fy * alpha. Then: x += vx, y += vy. Damping: vx *= 0.6, vy *= 0.6
       - Alpha *= ALPHA_DECAY
       - Return true if alpha < ALPHA_MIN (converged)

       `pub fn run_ticks(input: &LayoutInput) -> LayoutOutput`:
       - Clone nodes from input
       - Run `ticks` iterations of simulation_tick
       - Return positions and converged flag

       #[reactor]
       pub async fn ForceLayoutWorker(mut scope: ReactorScope<LayoutInput, LayoutOutput>) {
           while let Some(input) = scope.next().await {
               let output = forces::run_ticks(&input);
               scope.send(output);
           }
       }

       use std::cell::RefCell;
       use std::rc::Rc;
       use gloo_worker::reactor::ReactorBridge;
       use resyn_worker::{ForceLayoutWorker, LayoutInput, LayoutOutput};

       pub struct WorkerBridge {
           bridge: ReactorBridge<ForceLayoutWorker>,
       }

       impl WorkerBridge {
           pub fn new(callback: impl Fn(LayoutOutput) + 'static) -> Self {
               let bridge = ForceLayoutWorker::spawner()
                   .callback(callback)
                   .spawn("./resyn_worker.js");
               Self { bridge }
           }

           pub fn send(&self, input: LayoutInput) {
               self.bridge.send(input);
           }
       }
