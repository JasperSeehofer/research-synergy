       ```rust
       let seed_idx = graph.node_indices().next(); // first node = seed
       let mut depths: std::collections::HashMap<petgraph::graph::NodeIndex, u32> = std::collections::HashMap::new();
       if let Some(seed) = seed_idx {
           use std::collections::VecDeque;
           let mut queue = VecDeque::new();
           queue.push_back((seed, 0u32));
           depths.insert(seed, 0);
           while let Some((node, depth)) = queue.pop_front() {
               for neighbor in graph.neighbors(node) {
                   if !depths.contains_key(&neighbor) {
                       depths.insert(neighbor, depth + 1);
                       queue.push_back((neighbor, depth + 1));
                   }
               }
           }
       }
       ```
