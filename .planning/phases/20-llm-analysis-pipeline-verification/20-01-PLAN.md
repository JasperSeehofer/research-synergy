```rust
pub analysis_stage: Option<String>,
```
This field is `None` for all crawl events (backwards compatible). For analysis events, it carries one of:
- `"extracting"` — ar5iv text extraction stage
- `"nlp"` — TF-IDF corpus analysis stage
- `"llm"` — LLM paper annotation stage
- `"gaps"` — contradiction + ABC-bridge detection stage

The `event_type` field carries `"analysis_extracting"`, `"analysis_nlp"`, `"analysis_llm"`, `"analysis_gaps"`, `"analysis_complete"`, or `"analysis_error"` for analysis events (per D-03, RESEARCH Pattern 2).

Add `#[serde(default)]` on the `analysis_stage` field so existing serialized events without this field deserialize correctly.

**Step 2: Create StartAnalysis server function** in `resyn-app/src/server_fns/analysis.rs`:

Follow the `StartCrawl` pattern exactly (per RESEARCH Pattern 1):

```rust
use leptos::prelude::*;

#[server(StartAnalysis, "/api")]
pub async fn start_analysis() -> Result<String, ServerFnError> {
    #[cfg(feature = "ssr")]
    {
        use resyn_core::datamodels::progress::ProgressEvent;
        use std::sync::Arc;
        use tokio::sync::broadcast;
        use tracing::{error, info, warn};

        let db = use_context::<Arc<resyn_core::database::client::Db>>()
            .ok_or_else(|| ServerFnError::new("Database not available"))?;
        let tx = use_context::<broadcast::Sender<ProgressEvent>>()
            .ok_or_else(|| ServerFnError::new("Progress channel not available"))?;

        tokio::spawn(async move {
            let started = std::time::Instant::now();

            // Helper to broadcast analysis progress
            let send_progress = |tx: &broadcast::Sender<ProgressEvent>, event_type: &str, stage: &str, found: u64, pending: u64| {
                let _ = tx.send(ProgressEvent {
                    event_type: event_type.to_string(),
                    papers_found: found,
                    papers_pending: pending,
                    papers_failed: 0,
                    current_depth: 0,
                    max_depth: 0,
                    elapsed_secs: started.elapsed().as_secs_f64(),
                    current_paper_id: None,
                    current_paper_title: None,
                    analysis_stage: Some(stage.to_string()),
                });
            };

            // Read LLM provider from env (per D-06)
            let llm_provider = std::env::var("RESYN_LLM_PROVIDER").ok();

            // Build AnalyzeArgs from env
            let args = resyn_server::commands::analyze::AnalyzeArgs {
                db: String::new(), // not used — we pass db directly
                llm_provider: llm_provider.clone(),
                llm_model: std::env::var("RESYN_LLM_MODEL").ok(),
                force: false,
                full_corpus: false,
                verbose: false,
            };

            // Broadcast start
            send_progress(&tx, "analysis_extracting", "extracting", 0, 0);

            // Call the pipeline — but wrap to catch panics from std::process::exit
            // The pipeline has std::process::exit calls that would kill the server.
            // We call the individual stage functions that DON'T exit, or catch the result.
            match resyn_server::commands::analyze::run_analysis_pipeline(
                &db, args, 3, false
            ).await {
                Ok(()) => {
                    send_progress(&tx, "analysis_complete", "complete", 0, 0);
                    info!("Analysis pipeline completed successfully");
                }
                Err(e) => {
                    error!(error = %e, "Analysis pipeline failed");
                    let _ = tx.send(ProgressEvent {
                        event_type: "analysis_error".to_string(),
                        papers_found: 0,
                        papers_pending: 0,
                        papers_failed: 0,
                        current_depth: 0,
                        max_depth: 0,
                        elapsed_secs: started.elapsed().as_secs_f64(),
                        current_paper_id: None,
                        current_paper_title: None,
                        analysis_stage: Some("error".to_string()),
                    });
                }
            }
        });

        Ok("Analysis started".to_string())
    }
    #[cfg(not(feature = "ssr"))]
    unreachable!()
}
```
