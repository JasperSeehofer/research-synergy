**viewport_fit.rs — Add compute_single_node_pan_target:**

Add a new function after `compute_fit_target`:

```rust
/// Compute viewport target to center a single node without changing scale.
/// Clamps scale to [0.5, 2.0] range for comfortable viewing context.
/// Returns None if the node is not found in the node list.
pub fn compute_single_node_pan_target(
    nodes: &[NodeState],
    paper_id: &str,
    css_width: f64,
    css_height: f64,
    current_scale: f64,
) -> Option<FitAnimState> {
    let node = nodes.iter().find(|n| n.id == paper_id)?;
    let target_scale = current_scale.clamp(0.5, 2.0);
    let target_offset_x = css_width / 2.0 - node.x * target_scale;
    let target_offset_y = css_height / 2.0 - node.y * target_scale;
    Some(FitAnimState {
        active: true,
        target_scale,
        target_offset_x,
        target_offset_y,
    })
}
```

Per RESEARCH.md Pitfall 4: do NOT use `compute_fit_target` for single-node pan — it would zoom to max scale and lose graph context. Instead, preserve current scale (clamped to [0.5, 2.0]) and just center the offset on the target node.

```rust
// Search pulse glow ring (D-06)
if let (Some(ref highlighted_id), Some(pulse_start)) =
    (&state.search_highlighted, state.pulse_start_frame)
{
    if node.id == *highlighted_id {
        let elapsed = state.frame_counter.saturating_sub(pulse_start);
        let t = (elapsed as f64) / 120.0; // 120 frames = ~2s at 60fps
        if t < 1.0 {
            let alpha = (1.0 - t).max(0.0);
            let pulse_offset = 8.0 * (t * 3.0 * std::f64::consts::PI).sin();

            // Outer ring 1: pulsing
            self.ctx.begin_path();
            self.ctx
                .arc(
                    node.x,
                    node.y,
                    node.radius + 4.0 + pulse_offset,
                    0.0,
                    std::f64::consts::TAU,
                )
                .unwrap();
            self.ctx.set_stroke_style_str("#58a6ff");
            self.ctx.set_global_alpha(alpha * 0.8);
            self.ctx.set_line_width(2.0);
            self.ctx.stroke();

            // Outer ring 2: steady glow
            self.ctx.begin_path();
            self.ctx
                .arc(
                    node.x,
                    node.y,

                    0.0,

                .unwrap();

            self.ctx.set_line_width(1.0);

/// Highlights the first case-insensitive match of `query` in `text` using <strong> with accent color.
/// Returns the text as a view with the match highlighted.
fn highlight_text(text: &str, query: &str) -> impl IntoView {
    if query.is_empty() {
        return view! { <span>{text.to_string()}</span> }.into_any();
    }
    let lower_text = text.to_lowercase();
    let lower_query = query.to_lowercase();
    if let Some(start) = lower_text.find(&lower_query) {
        let before = text[..start].to_string();
        let matched = text[start..start + query.len()].to_string();
        let after = text[start + query.len()..].to_string();
        view! {
            <span>
                {before}
                <strong class="filter-match">{matched}</strong>
                {after}
            </span>
        }
        .into_any()
    } else {
        view! { <span>{text.to_string()}</span> }.into_any()
    }
}
```
