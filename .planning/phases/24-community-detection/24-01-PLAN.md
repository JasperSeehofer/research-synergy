    use std::collections::HashMap;

    /// A partitioning result that includes the size-ranked color_index for each community.
    #[derive(Debug, Clone, PartialEq)]
    pub struct LouvainPartition {
        /// arxiv_id -> community_id (stable Louvain id; may be sparse)
        pub assignments: HashMap<String, u32>,
        /// community_id -> color_index (0 = largest non-Other; OTHER_COLOR_INDEX for the Other bucket)
        pub color_indices: HashMap<u32, u32>,
        /// community_id -> member count

    /// Runs Louvain on the undirected projection of `graph`.
    /// Communities with < MIN_COMMUNITY_SIZE nodes are merged into a single "Other" community
    /// whose community_id is the numerically-largest assigned id and whose color_index is
    /// OTHER_COLOR_INDEX. Remaining communities are ranked by size descending and assigned
    /// color_index starting at 0. Returns a LouvainPartition with empty assignments when
    /// graph has < MIN_GRAPH_NODES nodes.
    pub fn detect_communities(
        graph: &StableGraph<Paper, f32, Directed, u32>,
    ) -> LouvainPartition;

    /// c-TF-IDF: each community is a pseudo-document.
    /// Input: community_id -> PaperAnalysis refs belonging to the community.
    /// Output: community_id -> Vec<(term, score)> sorted desc, truncated to TOP_KEYWORDS_PER_COMMUNITY.
    /// Skips communities whose color_index is OTHER_COLOR_INDEX.
    pub fn compute_ctfidf(
        community_members: &HashMap<u32, Vec<&PaperAnalysis>>,
        color_indices: &HashMap<u32, u32>,
    ) -> HashMap<u32, Vec<(String, f32)>>;

    /// Hybrid score from D-19: pagerank * (intra_community_degree + 1).
    pub fn hybrid_score(pagerank: f32, intra_degree: usize) -> f32 {
        pagerank * (intra_degree as f32 + 1.0)

    /// Top-N papers in a community, ranked by hybrid score descending.
    pub fn build_top_papers(
        community_members: &[&Paper],
        pagerank_by_id: &HashMap<String, f32>,
        intra_degree_by_id: &HashMap<String, usize>,
    ) -> Vec<CommunityTopPaper>;

    /// Build the auto-label for a community (top 1-2 c-TF-IDF keywords joined with space).
    /// Other bucket returns "Other". Empty keyword list returns "Unlabeled".
    pub fn build_community_label(color_index: u32, keywords: &[(String, f32)]) -> String;

    Louvain integration (module-private):
    1. Sort papers by `strip_version_suffix(&paper.id)` to guarantee deterministic indexing (Pitfall 2).
    2. Build a sorted arxiv_id -> u32 node index.
    3. Iterate `graph.edge_indices()`, remap endpoints to sorted indices, deduplicate as (min, max) pairs (undirected projection, D-02).
    4. `let network = Louvain::build_network(n_nodes, edges.len(), edges.into_iter());`
    5. `let mut louvain = Louvain::new(LOUVAIN_RESOLUTION, Some(LOUVAIN_SEED));`
    6. `let mut clustering = VectorGrouping::new(n_nodes);`
    7. `louvain.iterate(&network, &mut clustering);`
    8. Read community ids per node via `clustering.group_of(idx)` — verify the accessor against single-clustering source on first compile; use the correct method name if it differs.
    9. Compute size per raw community id.
    10. Communities with size < MIN_COMMUNITY_SIZE are relabeled to a single "Other" community_id (pick `u32::MAX - 1` or similar sentinel — avoid clashing with OTHER_COLOR_INDEX which is used only for color).
    11. Build `color_indices`: sort surviving non-Other community_ids by (size desc, community_id asc) and assign 0..N; the Other bucket gets OTHER_COLOR_INDEX (u32::MAX).
    12. Return `LouvainPartition { assignments, color_indices, sizes }`.

    c-TF-IDF (Pattern 2 from RESEARCH.md):
    - For each non-Other community: `tf_c(t) = sum(tfidf[t] for p in community) / community.len()`.
    - DF across non-Other communities: count of communities where `tf_c(t) > 0`.
    - IDF: `ln(n_communities / df(t)) + 1`.
    - Score: `c_tfidf(t, c) = tf_c(t) * idf(t)`.
    - Return top TOP_KEYWORDS_PER_COMMUNITY per community, sorted desc.

    Orchestrator steps for `compute_and_store_communities`:
    1. `PaperRepository::get_all_papers()`.
    2. `let fingerprint = corpus_fingerprint(&sorted_ids)`.
    3. `create_graph_from_papers(&papers)` -> StableGraph.
    4. `detect_communities(&graph)`. If `assignments.is_empty()`, call `CommunityRepository::delete_stale(&fingerprint)` and return `Skipped`.
    5. Bulk-load `PaperAnalysis` and `GraphMetrics` in single SurrealDB queries (Phase 23 pattern — RESEARCH.md forbids per-paper N+1).
    6. Build `Vec<CommunityAssignment>` for every paper (including those in the Other bucket — the Other bucket has a distinct community_id shared across all small-community members).
    7. `CommunityRepository::delete_stale(&fingerprint)` then `upsert(&assignments)`.
    8. Return `Computed { n_communities: non_other_count, n_papers: papers.len(), fingerprint }`.

    `compute_community_summaries` steps:
    1. Load all CommunityAssignment rows matching the current corpus_fingerprint.
    2. Load all PaperAnalysis + Paper + GraphMetrics rows in bulk.
    3. Group assignments by community_id. Reconstruct `LouvainPartition.color_indices` from the stored data (re-derive by size rank — produces the same values because Plan 01 is deterministic).
    4. Compute intra-community degrees by traversing only the edges whose endpoints are both in the same community (undirected projection).
    5. Call `compute_ctfidf` across all communities (cross-community scope is required for distinctiveness).
    6. For each community: call `build_top_papers`, call `shared_high_weight_terms` on member TF-IDF vectors, call `build_community_label` with the top keywords. Assemble `CommunitySummary`.
    7. Sort returned vec by (color_index ascending for non-Other, Other last).

    `load_community_status`:
    - Query `PaperRepository::get_all_papers()` to compute the current `fingerprint`.
    - Query `CommunityRepository::list_all()`; if empty -> `ready: false, fingerprint: None, count: 0`.
    - If any row's corpus_fingerprint matches `fingerprint` -> `ready: true, fingerprint: Some(fingerprint), count: <distinct non-Other community ids>`.
    - Otherwise -> `ready: false, fingerprint: None, count: 0` (stale cache is treated as not-ready from the UI's perspective).
